# drop-in here — the code only uses yaml.safe_load / yaml.dump.

# paho-mqtt 2.x changed the Client() constructor to require a callback API
# version; the clients pass CallbackAPIVersion.VERSION1 to keep the 1.x
# callback signatures (see shedder/integration/mqtt.py, ocpp/mqtt_client.py).
paho-mqtt==2.1.0
ocpp==2.1.0
# websockets 14+ makes the new asyncio server API the default (websocket.request,
# single-arg handler) which the OCPP server already uses; ocpp 2.1.0 no longer
# constrains websockets, so the old <14 cap is dropped.
websockets==16.0
